print("\nPreprocessing Data...")
# Cleaning - Numeric columns pe mean lagana
numeric_cols_diabetes = diabetes_data.select_dtypes(include=[np.number]).columns
diabetes_data[numeric_cols_diabetes] = diabetes_data[numeric_cols_diabetes].fillna(diabetes_data[numeric_cols_diabetes].mean())

numeric_cols_heart = heart_data.select_dtypes(include=[np.number]).columns
heart_data[numeric_cols_heart] = heart_data[numeric_cols_heart].fillna(heart_data[numeric_cols_heart].mean())
heart_data = heart_data.dropna()

# Combine datasets with common features
common_features = ['Age', 'Glucose', 'BloodPressure', 'BMI', 'Outcome', 'Disease']
diabetes_subset = diabetes_data[['Age', 'Glucose', 'BloodPressure', 'BMI', 'Outcome', 'Disease']]

# Heart dataset mein 'Glucose' nahi hai, toh dummy value (mean) daalenge
heart_subset = heart_data[['Age', 'RestBP', 'OldPeak', 'Outcome', 'Disease']].rename(columns={'RestBP': 'BloodPressure', 'OldPeak': 'BMI'})
heart_subset['Glucose'] = diabetes_data['Glucose'].mean()  # Placeholder value for Glucose
heart_subset = heart_subset[['Age', 'Glucose', 'BloodPressure', 'BMI', 'Outcome', 'Disease']]  # Reorder columns

combined_data = pd.concat([diabetes_subset, heart_subset], ignore_index=True)

# Feature Engineering
combined_data['Glucose_BP_Ratio'] = combined_data['Glucose'] / (combined_data['BloodPressure'] + 1)
combined_data['BMI_Age_Interaction'] = combined_data['BMI'] * combined_data['Age']

# Correlation Heatmap
plt.figure(figsize=(10, 6))
sns.heatmap(combined_data.drop('Disease', axis=1).corr(), annot=True, cmap='coolwarm')
plt.title("Feature Correlation Heatmap")
plt.show()

print("Preprocessing Done!")
print(combined_data.head())

print("\nTraining Deep Learning Model...")
model = Sequential([
    Dense(64, activation='relu', input_shape=(X_train.shape[1],)),
    Dropout(0.3),
    Dense(32, activation='relu'),
    Dropout(0.2),
    Dense(16, activation='relu'),
    Dense(1, activation='sigmoid')
])

model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
history = model.fit(X_train, y_train, epochs=50, batch_size=32, validation_split=0.2, verbose=0)

# Evaluate
dl_pred = (model.predict(X_test) > 0.5).astype(int)
dl_accuracy = accuracy_score(y_test, dl_pred)
print(f"Deep Learning Accuracy: {dl_accuracy * 100:.2f}%")

def predict_disease(patient_data):
    patient_data = np.array(patient_data).reshape(1, -1)
    patient_data_scaled = scaler.transform(patient_data)
    rf_prob = rf_model.predict_proba(patient_data_scaled)[0]
    dl_prob = model.predict(patient_data_scaled, verbose=0)[0][0]
    rf_result = "Positive" if rf_model.predict(patient_data_scaled)[0] == 1 else "Negative"
    dl_result = "Positive" if dl_prob > 0.5 else "Negative"
    return rf_result, rf_prob, dl_result, dl_prob
